## 交付物模板 3：样本量计算

```python
# sample_size_calculator.py
from scipy import stats

def calculate_sample_size(
    baseline_rate: float,    # 基线转化率
    mde: float,              # 最小可检测效应（相对）
    alpha: float = 0.05,     # 显著性水平
    power: float = 0.8,      # 统计功效
    ratio: float = 1.0       # 实验组/对照组比例
) -> int:
    """
    计算 A/B 测试所需样本量
    
    示例：
    - 基线转化率 10%
    - 期望检测到 15% 相对提升（即 10% → 11.5%）
    - 显著性 5%，功效 80%
    - 结果：每组约 12,000 用户
    """
    p1 = baseline_rate
    p2 = baseline_rate * (1 + mde)
    
    z_alpha = stats.norm.ppf(1 - alpha / 2)
    z_beta = stats.norm.ppf(power)
    
    pooled_p = (p1 + p2) / 2
    
    n = (
        (z_alpha * (2 * pooled_p * (1 - pooled_p)) ** 0.5 +
         z_beta * (p1 * (1 - p1) + p2 * (1 - p2)) ** 0.5) ** 2
    ) / (p2 - p1) ** 2
    
    return int(n * (1 + ratio) / ratio)

# 使用示例
n = calculate_sample_size(
    baseline_rate=0.10,
    mde=0.15,  # 15% 相对提升
    alpha=0.05,
    power=0.8
)
print(f"每组需要样本量: {n}")  # 输出: 每组需要样本量: 12342
```
